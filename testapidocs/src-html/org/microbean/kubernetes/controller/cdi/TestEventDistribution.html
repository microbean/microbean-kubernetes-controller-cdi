055@ApplicationScoped
056public class TestEventDistribution {
057
058  
059  /*
060   * Static fields.
061   */
062
063  
064  /**
065   * The number of instances of this class that have been created (in
066   * the context of JUnit execution; any other usage is undefined).
067   */
068  private static int instanceCount;
069
070
071  /*
072   * Constructors.
073   */
074
075  public TestEventDistribution() {
076    super();
077    instanceCount++;
078  }
079
080  @Produces
081  @ApplicationScoped
082  @AllConfigMapEvents
083  private static final Operation<ConfigMap, ConfigMapList, DoneableConfigMap, Resource<ConfigMap, DoneableConfigMap>> selectAllConfigMaps(final KubernetesClient client) {
084    return client.configMaps();
085  }
086
087  private final void onConfigMapSynchronizationAddition(@ObservesAsync @AllConfigMapEvents @Addition(synchronization = true) final ConfigMap configMap) {
088    assertNotNull(configMap);
089  }
090  
091  private final void onConfigMapModification(@ObservesAsync @AllConfigMapEvents @Modification final ConfigMap configMap, @Prior final Optional<ConfigMap> prior) {
092    assertNotNull(configMap);
093    org.microbean.cdi.AbstractBlockingExtension.unblockAll();
094  }
095
096  private final void onStartup(@Observes @Initialized(ApplicationScoped.class) final Object event, final KubernetesControllerExtension extension) throws InterruptedException {
097    assertNotNull(extension);
098  }
099  
100  @Test
101  public void testContainerStartup() {
102    assumeFalse(Boolean.getBoolean("skipClusterTests"));
103    System.setProperty("port", "8080");
104    final int oldInstanceCount = instanceCount;
105    assertEquals(1, oldInstanceCount);
106    Main.main(null);
107    assertEquals(oldInstanceCount + 1, instanceCount);
108  }
109
110  @BeforeClass
111  public static void setUpLogging() {
112    System.setProperty("java.util.logging.config.file", Thread.currentThread().getContextClassLoader().getResource("logging.properties").getPath());
113  }
114
115  @Documented
116  @KubernetesEventSelector
117  @Qualifier
118  @Retention(value = RetentionPolicy.RUNTIME)
119  @Target({ ElementType.METHOD, ElementType.PARAMETER })
120  private @interface AllConfigMapEvents {
121
122  }
123  
124}
